const Weddings = require('../models/wedding');

function indexRoute(req, res, next) {
  Wedding
    .find()
    .exec()
    .then((weddings) => res.render('weddings/index', { weddings }))
    .catch(next);
}

function newRoute(req, res) {
  return res.render('weddings/new');
}

function createRoute(req, res, next) {
  console.log(req.body);
  req.body.createdBy = req.user;
  Wedding
    .create(req.body)
    .then(() => res.redirect('/weddings'))
    .catch((err) => {
      if(err.name === 'ValidationError') {
        return res.badRequest('/weddings/new', err.toString());
      }
      next(err);
    });
}

function showRoute(req, res, next) {
  Wedding
    .findById(req.params.id)
    .populate('createdBy')
    .exec()
    .then((wedding) => {
      if(!wedding) return res.notFound();
      return res.render('weddings/show', { wedding });
    })
    .catch(next);
}

function editRoute(req, res, next) {
  Wedding
    .findById(req.params.id)
    .exec()
    .then((wedding) => {
      if(!wedding) return res.notFound();
      return res.render('weddings/edit', { wedding });
    })
    .catch(next);
}

function updateRoute(req, res, next) {
  Wedding
    .findById(req.params.id)
    .exec()
    .then((wedding) => {
      if(!wedding) return res.notFound();

      wedding = Object.assign(wedding, req.body);

      return wedding.save();
    })
    .then(() => res.redirect(`/weddings/${req.params.id}`))
    .catch((err) => {
      if(err.name === 'ValidationError') {
        return res.badRequest(`/weddings/${req.params.id}/edit`, err.toString());
      }
      next(err);
    });
}

function deleteRoute(req, res, next) {
  Wedding
    .findById(req.params.id)
    .exec()
    .then((wedding) => {
      if(!wedding) return res.notFound();
      return wedding.remove();
    })
    .then(() => res.redirect('/weddings'))
    .catch(next);
}

function deleteCommentRoute(req, res, next) {
  Wedding
    .findById(req.params.id)
    .exec()
    .then((wedding) => {
      if(!wedding) return res.notFound();

      const comment = wedding.comments.id(req.params.commentId);
      comment.remove();

      return wedding.save();
    })
    .then((wedding) => {
      res.redirect(`/weddings/${wedding.id}`);
    })
    .catch(next);
}

function createCommentRoute(req, res, next) {

// attach the logged in user to the body of the request
  req.body.createdBy = req.user;

  Wedding
    .findById(req.params.id)
    .exec()
    .then((wedding) => {
      if(!wedding) return res.notFound();

      wedding.comments.push(req.body);
      return wedding.save();
    })
    .then((wedding) => {
      res.redirect(`/weddings/${wedding.id}`);
    })
    .catch(next);
}

module.exports = {
  index: indexRoute,
  new: newRoute,
  create: createRoute,
  show: showRoute,
  edit: editRoute,
  update: updateRoute,
  delete: deleteRoute,
  createComment: createCommentRoute,
  deleteComment: deleteCommentRoute
};
